pragma solidity 0.5.1;

contract IceCreamShopContract {
    
    //Количество магазинов
    uint256 public shopCount;
    mapping (uint=>IceCreamShop) public shop;
    
    
    struct IceCreamShop {
        uint id;
        string name;
        //Площадь магазина
        uint square;
        //Стоимость мороженого
        uint iceCreamCost;
        //Количество мороженого
        uint iceCreamAmount;
        //Наличие мороженого в магазине
        bool availabilityIceCream;
    }
    
    function addShop (string memory name, uint square, uint iceCreamCost, uint iceCreamAmount) public {
        //Увеличиваем кол-во магазинов
        shopCount+=1;
        bool _availabilityIceCream;
        
        //Если количество мороженого равно 0 значит ставим false в наличии
        if (iceCreamAmount==0){
            _availabilityIceCream = false;
        }
        else _availabilityIceCream = true;
        
        //Добавляем магазин
        shop[shopCount] = IceCreamShop(shopCount,name, square, iceCreamCost, iceCreamAmount, _availabilityIceCream);
    }

}
