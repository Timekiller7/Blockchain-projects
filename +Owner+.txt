pragma solidity 0.5.1;

contract test_3 {
uint256 peopleCount=0;
mapping(uint=>Person) public people;
address owner;
modifier onlyOwner(){
require (msg.sender==owner);
_;
}

struct Person {uint id; string firstName; string lastName;}

constructor() public {                     //по умолчанию адрес владельца устанавливается
owner=msg.sender;
}

function addPerson (string memory firstName, string memory lastName) public onlyOwner {
incrementCount();
people[peopleCount]=Person (peopleCount, firstName, lastName);
}

function incrementCount() internal {
peopleCount+=1;
}
}
